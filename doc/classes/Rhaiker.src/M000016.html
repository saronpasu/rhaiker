  def friendship_create(user_id)
    uri = create_uri(
      BaseAddress +
      'friendships/create/' +
      user_id + '.xml'
    )
    request = create_request(:post, uri, true)
    document = http_access(uri, request)
    if document
      return parse_user(document.root)
    else
      return false
    end
  end
