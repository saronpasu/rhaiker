  def keyword_relation_create(word1, word2)
    uri = create_uri(
      BaseAddress +
      'keywords/relation/create.xml' +
      '?word1=' + word1 +
      '&word2=' + word2
    )
    request = create_request(:post, uri, true)
    document = http_access(uri, request)
    if document
      return parse_keyword(document.root)
    else
      return false
    end
  end
