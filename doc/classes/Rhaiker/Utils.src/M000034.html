    def parse_options(options)
      result = ''
      if options[:since]
        options[:since] = Time.parse(options[:since].to_s).httpdate
      end
      options = options.find_all{|option|option.last}
      result = '?' + options.collect{|option|
        option.join('=')
      }.join('&') unless options.empty?
      return result
    end
